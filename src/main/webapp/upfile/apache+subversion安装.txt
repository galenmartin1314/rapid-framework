1.解压apache执行如下命令安装
./configure --prefix=/develop/apache  --with-apr=/usr/local/apr/bin/apr-1-config --with-apr-util=/usr/local/apr/bin/apu-1-config --enable-so --enable-mods-shared=all --enable-modules=all --enable-dav --enable-dav-fs --enable-maintainer-mode --enable-rewrite
make
make isntall
2.解压subversion
./configure --prefix=/develop/subversion --with-apxs=/develop/apache/bin/apxs --with-apr=/usr/local/apr/bin/apr-1-config --with-apr-util=/usr/local/apr/bin/apu-1-config --with-ssl --with-zlib --enable-maintainer-mode --without-berkeley-db
make
make install
3.若/develop/apache/modules目录中不存在mod_dav_svn.so、mod_authz_svn.so这两个文件则从subversion安装目录中拷贝
/develop/subversion-1.7.13/subversion/mod_dav_svn/.libs/mod_dav_svn.so
/develop/subversion-1.7.13/subversion/mod_authz_svn/.libs/mod_authz_svn.so
4.集成apache和subversion
修改apache/conf/httpd.conf
在末尾添加
<Location /svn>
DAV svn
SVNlistParentPath on
SVNParentPath /develop/svnroot
#SVNPath /var/local/test
AuthzSVNAccessFile /develop/svnroot/authz.conf
AuthType Basic
AuthName "Subversion"
AuthUserFile /develop/svnroot/passwd.conf
Require valid-user
</Location>
加载库文件
LoadModule dav_svn_module modules/mod_dav_svn.so 
LoadModule authz_svn_module modules/mod_authz_svn.so
修改为
User apache
Group apache
5.重启apache
6.访问 http://ip:port/svn/资源库
